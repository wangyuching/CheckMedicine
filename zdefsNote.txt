def get_target_obb_cls(results, target_cls):
    filtered_boxes = []
        初始化一個空列表，用來存放篩選後的框

    for r in results:
        遍歷模型回傳的每一個偵測結果
        classes = r.obb.cls
            取得所有偵測到物件的class ID
        boxes = r.obb.xywhr #[center_x, center_y, width, height, rotation_radians]
            取得所有物件的旋轉框(obb)座標，格式為 [中心x, 中心y, 寬, 高, 旋轉弧度]

        mask = (classes == target_cls)
            建立布林遮罩 (Boolean Mask)，當類別等於目標類別時為 True，否則為 False
        target_boxes = boxes[mask]
            利用遮罩，只保留符合目標類別的旋轉框
        target_boxes_np = target_boxes.cpu().numpy()
            將資料移至 CPU 記憶體並轉換成 NumPy 陣列格式
        
        for box in target_boxes_np:
            filtered_boxes.append(box)
                將每一個符合條件的框加入到結果列表中
            
    return filtered_boxes
        回傳所有篩選出來的旋轉框列表


def pillbox_head_tail(bedtime_word, pill_box):

    如果沒偵測到睡前文字，回傳 False
    if not bedtime_word:
        return False
    
    px, py, pw, ph, pr = pill_box
        分離藥盒的座標與角度資訊
    bx, by, bw, bh, br = bedtime_word[0]
        取得睡前文字的座標資訊

        根據藥盒長寬比，判斷藥盒是橫放 (pw > ph) 還是直放
    if pw > ph: 橫放。
        axis_vec = np.array([np.cos(pr), np.sin(pr)])
            計算藥盒主軸（寬度方向）的單位方向向量。
    else: 直放
        axis_vec = np.array([-np.sin(pr), np.cos(pr)])
            計算藥盒主軸（高度方向）的單位方向向量

    target_vec = np.array([bx - px, by - py])
        建立一個從「藥盒中心」指向「睡前文字中心」的向量

    projection = np.dot(target_vec, axis_vec)
        計算兩個向量的內積（點積），這代表睡前文字在藥盒主軸上的投影位置

    return True if projection < 0 else False
        如果投影值(內積)小於 0，代表文字在主軸的負半軸，回傳 True；否則回傳 False
        內積 > 0（正數）：代表兩個向量的夾角小於 90 度（朝向同一個大方向）。
                文字位於主軸正方向的那一端（Tail）。
        內積 < 0（負數）：代表兩個向量的夾角大於 90 度（朝向相反方向）。
                文字位於主軸反方向的那一端（Head）。


def split_pillbox_to_slots(obb_xywhr, axis='w', num_splits=4, reverse=False):
    xc, yc, w, h, r = obb_xywhr
        xc, yc 為藥盒中心點；w, h 為寬高；r 為旋轉弧度 (Radians)

        計算切分後，每一個小格子的新寬度或新高度
    new_w = w / num_splits if axis == 'w' else w
        如果切分軸向 axis 是 'w'（代表要水平切分寬度）：
        新的寬度 new_w 就會是原本的寬度 w 除以(float)切分份數 num_splits。
        否則（例如 axis 是 'h' 或者是其他值）：
        寬度保持不變，new_w 直接等於原本的寬度 w。
        
    new_h = h / num_splits if axis == 'h' else h    
        如果切分軸向 axis 是 'h'（代表要垂直切分高度）：
        新的高度 new_h 就會是原本的高度 h 除以(float)切分份數 num_splits。
        否則（例如 axis 是 'w'）：
        高度保持不變，new_h 直接等於原本的高度 h。
    
    slots = []
        初始化存放各個小格子座標的列表
    
        產生num_splits份數得均勻分佈的小區塊中心點座標的比例值。例如分成 4 份，會產生 [-0.375, -0.125, 0.125, 0.375] 的相對位置
    steps = np.linspace(-0.5 + 1/(2*num_splits), 0.5 - 1/(2*num_splits), num_splits)
        np.linspace(start, stop, num) 在指定的 start 到 stop 之間，均勻生成 num 個數字。
        -0.5：最左側的邊界。+1/(2*num_splits)：座標「往右前進」到第一格中心。
        0.5：最右側的邊界。-1/(2*num_splits)：座標「往左後退」到最後一格中心。

    if reverse:
        steps = steps[::-1]
            如果設定反向（reverse=True），就將格子的順序倒序排列（例如從右到左，或從下到上）

        計算出每個小格子的中心點座標。
    for step in steps:
        # 根據切分方向（沿著寬 w 或 高 h），計算小格子相對於藥盒中心的位移量(沒旋轉)
        if axis == 'w':
            dx, dy = step * w, 0
                沿著寬度方向位移
        else:
            dx, dy = 0, step * h
                沿著高度方向位移
            
            【旋轉矩陣】考量到藥盒本身有旋轉角度 r，把藥盒的中心點 (xc, yc) 當作原點，
                        加上旋轉修正後的位移量，映射回影像的絕對座標 (new_x, new_y)。
        new_x = xc + dx * np.cos(r) - dy * np.sin(r)
        new_y = yc + dx * np.sin(r) + dy * np.cos(r)
        
        slots.append([new_x, new_y, new_w, new_h, r])
            將計算好的小格子座標 [新中心x, 新中心y, 新寬, 新高, 旋轉弧度] 存入列表
        
    return slots
        回傳所有切分好的小格子座標列表

        
    透過計算歐幾里得距離(兩點距離公式)，找出這個蓋子離哪一個格子最近。
def lid_connect_slots(lid_box, slots):
    lx, ly = lid_box[0], lid_box[1]
        取得藥盒蓋子的中心點座標 (lx, ly)
    min_dist = float('inf')
        初始化最小距離為正無窮大
    best_idx = -1
        初始化最匹配的小格子索引為 -1

    for idx, s_box in enumerate(slots):
        sx, sy = s_box[0], s_box[1]
            取得當前格子的中心點座標 (sx, sy)
        dist = np.sqrt((lx - sx)**2 + (ly - sy)**2)
            計算蓋子中心與格子中心之間的直線距離（歐XXX距離）
        if dist < min_dist:
            如果當前計算的距離比之前的最小值還要小
            min_dist = dist
                更新最小值
            best_idx = idx
                紀錄目前最接近的小格子索引值

    return best_idx
        回傳距離最近的小格子索引


def check_pill_in_slots(frame, i, box, hsv_lower, hsv_upper, threshold=0.1):
    xc, yc, w, h, r = box
        分離小格子的旋轉框參數
    M = cv2.getRotationMatrix2D((xc, yc), np.degrees(r), 1)
        根據格子中心與旋轉角度（弧度轉角度），建立 2D 旋轉矩陣
    rotated = cv2.warpAffine(frame, M, (frame.shape[1], frame.shape[0]))
        將整張影像進行仿射變換 (Warp Affine)，把傾斜的影像轉正
    base_crop = cv2.getRectSubPix(rotated, (int(w), int(h)), (xc, yc))
        從轉正的影像中，精確擷取特定大小與中心的矩形區域

    h_crop, w_crop = base_crop.shape[:2]
        取得裁剪影像的高度與寬度
    pad_w = int(w_crop * 0.05)
        計算寬度 5% 的邊界範圍
    pad_h = int(h_crop * 0.05)
        計算高度 5% 的邊界範圍
    crop = base_crop[pad_h:-pad_h, pad_w:-pad_w]
        去除四周邊緣 5% 的區域，避免格子邊框線干擾顏色判斷

    if (crop is None) or (crop.size == 0):
        return False, np.zeros((10, 10), dtype=np.uint8)
            如果裁剪失敗或區域為空，回傳無藥丸及空遮罩
    

    hsv_img = cv2.cvtColor(crop, cv2.COLOR_BGR2HSV)
        將 BGR 色彩空間轉換為 HSV，更有利於色彩過濾。
    base_mask = cv2.inRange(hsv_img, hsv_lower, hsv_upper)
        根據藥丸的 HSV 上下限建立遮罩（特定顏色變白色，其餘變黑色）

    kernel = np.ones((3, 3), np.uint8)
        建立 3x3 的結構元素（核心）
    mask = cv2.morphologyEx(base_mask, cv2.MORPH_CLOSE, kernel)
        進行型態學「閉運算」（先膨脹後侵蝕），填補遮罩中的小黑洞

    white_pixels = cv2.countNonZero(mask)
        計算遮罩中白色的像素點總數（代表偵測到的藥丸顏色面積）
    total_pixels = w * h
        計算格子的總像素面積
    ratio = white_pixels / total_pixels
        計算藥丸顏色所佔的比例

    has_pill = ratio> threshold
        如果比例大於設定的閾值（threshold=0.1(10%)），則判定「有藥丸（True）」

    # print(f"Box {i}: Pill Ratio = {ratio:.2%}")

    return has_pill, mask
        回傳是否有藥丸的布林值，以及處理後的二值化遮罩影像

def draw_slots_state(frame, box, slot_idx, slot_data):
    x, y,w, h, r = box
        分離格子座標
    lid_state = slot_data['lid']
        取得蓋子狀態（"Open"、"Close" 或其他）
    pill_state = "Full" if slot_data['Has_pill'] else "Empty"
        根據布林值設定藥丸狀態文字

    if lid_state == "Open" and not slot_data["Has_pill"]:
        color = (0, 255, 0)  
            # 蓋子打開且沒藥（代表已服藥）-> 顯示 綠色

    elif lid_state == "Open" and slot_data["Has_pill"]:
        color = (0, 0, 255)  # Red for open & full
            蓋子打開但藥還在（異常或剛打開）-> 顯示 紅色

    elif lid_state == "Close":
        color = (255, 0, 0)  # Blue for closed lid
            蓋子關閉中 -> 顯示 藍色

    else:
        color = (100, 100, 100)  # Gray for missing lid
            蓋子不見或未定義 -> 顯示 灰色

    rect = ((x, y), (w, h), np.degrees(r))
        建立 OpenCV 定義的旋轉矩形格式 ((中心), (寬高), 角度)
    points = cv2.boxPoints(rect)
        將旋轉矩形轉換成四個頂點的座標
    points = np.int32(points)
        將頂點座標轉換成整數型態
        
    cv2.polylines(frame, [points], isClosed=True, color=color, thickness=2)
        在影像上繪製這個旋轉框

    label = f"#{slot_idx} {lid_state}"
        組合蓋子編號與狀態文字，例如：#0 Open
    cv2.putText(frame, label, (int(x) - 30, int(y)), cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 1)
        在畫面上繪製格子編號與蓋子狀態

    if lid_state == "Open":
        如果蓋子是開的
        pill_label = f"{pill_state}"
            準備藥丸狀態標籤
        cv2.putText(frame, pill_label, (int(x) - 30, int(y) + 15), cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 1)    
            在原標籤下方 15 像素處，繪製藥丸狀態（Full / Empty）

def every_slots_state(frame, current_slots_data, tracker, duration, missing, db_insert):
        
        找出當前畫面上所有蓋子狀態為 "Open" 的格子索引列表
    current_opens = [
        idx for idx, 
        data in current_slots_data.items() 
        if data['lid'] == "Open"
    ]

        當前有格子是被打開的
    if len(current_opens) > 0:
        tracker['missing_start_time'] = None
            重置「消失計時器」，因為藥盒目前在畫面上且有開蓋

            如果「當前打開的格子」與「追蹤器記錄中打開的格子」不一致（代表開啟狀態改變了）
        if current_opens != tracker['active_opens']:
            tracker['active_opens'] = current_opens
                更新追蹤器中處於開啟狀態的格子列表
            tracker['open_start_time'] = t.time()
                重新開始計算開啟的起點時間
            tracker['triggered'] = False
                重置觸發旗標，表示這個新狀態尚未寫入資料庫

        else:如果開啟的格子與上次檢查完全相同（狀態穩定維持中）
                當有記錄開始時間，且該狀態還沒觸發過資料庫寫入時
            if tracker['open_start_time'] is not None and not tracker['triggered']:
                elapsed_time = t.time() - tracker['open_start_time']
                    計算已經持續開啟了多少秒

                    如果持續開啟的時間超過了設定的閾值（duration）
                if elapsed_time > duration:
                    db_insert(current_slots_data)
                        將當前的藥盒格子的詳細狀態寫入資料庫
                    tracker['triggered'] = True
                        將旗標設為 True，防止在同一次開蓋事件中重複寫入資料庫
    
    else:當前畫面上沒有任何格子被打開（可能全部關閉，或藥盒離開畫面沒被偵測到）
            如果在此之前，曾經有格子是打開的
        if tracker['open_start_time'] is not None:
            if tracker['missing_start_time'] is None:
                tracker['missing_start_time'] = t.time()
                    開始記錄「消失/關閉」的起點時間（容錯計時開始

            lost_duration = t.time() - tracker['missing_start_time']
                計算消失/關閉持續了多久

                如果消失或關閉的時間超過了容錯上限（missing）
            if lost_duration > missing:
                全面重置追蹤器的所有狀態，宣告這一次的開蓋事件正式結束
                tracker['active_opens'] = []
                    目將追蹤器中「目前處於開啟狀態的格子列表」清空
                tracker['open_start_time'] = None
                    將「開蓋起始時間」設定為空值
                tracker['missing_start_time'] = None
                    將「藥盒消失（或全關閉）的起始時間」設定為空值
                tracker['triggered'] = False
                    將「是否已觸發資料庫寫入」的旗標（Flag）重新設為「否（False）」。
